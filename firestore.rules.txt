rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    // Friends
    match /friends/{friendId} {
      allow read, write: if request.auth != null;
    }
    // Followers & Following
    match /followers/{id} { allow read, write: if request.auth != null; }
    match /following/{id} { allow read, write: if request.auth != null; }
    // Chats
    match /chats/{chatId} {
      allow read, write: if request.auth != null && request.auth.uid in resource.data.participants;
    }
    // Messages
    match /chats/{chatId}/messages/{messageId} {
      allow read, write: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;
    }
    // Stories
    match /stories/{storyId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.uid;
      allow delete: if request.auth != null && request.auth.uid == resource.data.uid;
      allow update: if request.auth != null && request.auth.uid == resource.data.uid;
    }
    // Story views
    match /storyViews/{viewId} { allow read, write: if request.auth != null; }
    // Groups
    match /groups/{groupId} {
      allow read, write: if request.auth != null && request.auth.uid in resource.data.members;
    }
    // Group members
    match /groups/{groupId}/members/{memberId} { allow read, write: if request.auth != null; }
    // Group messages
    match /groups/{groupId}/messages/{messageId} {
      allow read, write: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members;
    }
    // Notifications
    match /notifications/{notifId} {
      allow read: if request.auth != null && request.auth.uid == resource.data.to;
      allow create: if request.auth != null;
    }
    // Calls
    match /calls/{callId} { allow read, write: if request.auth != null; }
    // Typing & Presence
    match /typingStatus/{chatId} { allow read, write: if request.auth != null; }
    match /presence/{uid} { allow read, write: if request.auth != null; }
    // Blocked users
    match /blockedUsers/{id} { allow read, write: if request.auth != null; }
  }
}
