rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /profiles/{userId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId && request.resource.size < 5 * 1024 * 1024;
    }
    match /covers/{userId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /stories/{userId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId && request.resource.contentType.matches('image/.*|video/.*');
    }
    match /chat-images/{chatId}/{file} {
      allow read, write: if request.auth != null;
    }
    match /chat-videos/{chatId}/{file} {
      allow read, write: if request.auth != null;
    }
    match /voice-notes/{chatId}/{file} {
      allow read, write: if request.auth != null;
    }
    match /documents/{chatId}/{file} {
      allow read, write: if request.auth != null;
    }
    match /group-media/{groupId}/{file} {
      allow read, write: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members;
    }
  }
}
